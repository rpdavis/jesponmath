  Step 1: I have 8, I need 2 to make 10

  Step 4: 10 + 5 = 15 (easy!)

### Lesson Data Structure

```typescript
interface StrategyLesson {
  id: string                    // 'making-5', 'making-10', etc.
  title: string                 // 'Making 5'
  operation: OperationType      // 'addition'
  requiredAfterSession: number  // 1, 6, etc.
  prerequisites: string[]       // ['making-5'] for making-10
  
  // Lesson content
  overview: {
    what: string                // What is this strategy?
    why: string                 // Why is it important?
    when: string                // When to use it?
    examples: LessonExample[]   // Worked examples
    memoryTricks: string[]      // Mnemonics
  }
  
  // Video
  video: {
    provider: 'khan' | 'youtube'
    url: string
    startTime?: number          // Optional start timestamp
    duration: number            // Estimated watch time
  }
  
  // Practice
  practice: {
    type: 'missing-number' | 'drag-drop' | 'standard' | 'scaffolded'
    problems: LessonProblem[]
    successCriteria: {
      minCorrect: number        // e.g., 8 out of 10
      minPercentage: number     // e.g., 80%
    }
  }
  
  // Completion
  completionMessage: string
  nextSteps: string
}

interface LessonProblem {
  id: string
  problemText: string           // "4 + _ = 5"
  problemType: 'missing-addend' | 'standard' | 'scaffolded' | 'visual'
  correctAnswer: string
  hints: string[]               // Progressive hints
  visualAid?: {
    type: 'ten-frame' | 'number-line' | 'hands'
    interactive: boolean        // Can student interact?
  }
  feedbackCorrect: string       // "Great! 4 + 1 = 5"
  feedbackIncorrect: string     // "Not quite. 4 needs 1 to make 5"
}
```

async function checkForRequiredLesson(sessionNumber: number, studentUid: string) {
  const completedLessons = await getCompletedLessons(studentUid)
  
  // Check if student needs a lesson before this session
  if (sessionNumber === 2 && !completedLessons.includes('making-5')) {
    return 'making-5'  // Trigger Making 5 lesson
  }
  
  if (sessionNumber === 7 && !completedLessons.includes('making-10')) {
    // Check prerequisite
    if (completedLessons.includes('making-5')) {
      return 'making-10'  // Trigger Making 10 lesson
    }
  }
  
  // Check if student is ready for decomposing lesson
  const progress = await getFluencyProgress(studentUid)
  if (needsBridgingProblems(progress) && !completedLessons.includes('decomposing')) {
    return 'decomposing'  // Trigger Decomposing lesson
  }
  
  return null  // No lesson needed, continue to practice
}
